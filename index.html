        print(i)

number_1 = int(input("Enter number one: "))
number_2 = int(input("Enter number two: "))

even_numbers = []
odd_numbers = []

for number in range(number_1, number_2 + 1):
    if number % 2 == 0:
        even_numbers.append(number)
    else:
        odd_numbers.append(number)
